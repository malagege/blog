.then((jsonObj)=>{

     */ 

    jsonObj.forEach(data => {
        /////////(3)
        let mail_to = data.usermail
        let email_template = `
        Hello World!!!!${data.img}
        <img src="cid:00000001${data.img}"/><br>
        姓名：${data.username}<br>
        訂單編號：${data.no}<br>
        `

        let mailOptions = {
          from: mail_from,
          to: mail_to,     //寄送人
          subject: subject,
          html: email_template,
          attachments: [{
            filename: data.img,
            path: data.img,
            cid: '00000001'+data.img
          }]
        };

        if (error) {
          console.log('mail_to:' + mail_to)
          console.log(error);
        } else {
          console.log(mail_to + ' , Email sent: ' + info.response);
        }
      });
      
      });
