<?php
include './vendor/autoload.php';

# 第 1 步: 建立 DI container
$builder = new \DI\ContainerBuilder();
$container = $builder->build();

# 第 2 步: 建立 class 
$a = $container->get('A');

var_dump($a);

# 第 3 步: 建立 DI 實作 class，B class 不需要做 new 動作
class A{
    private $_b;
    function __construct(B $b)
    {
        $this->_b = $b;
        // $b->hello();
    }
}

class B{
    function __construct()
    {
        echo "Hello World! B";
    }
    function hello()
    {
        echo "B:say hello";
    }
}

<?php
include './vendor/autoload.php';


$builder = new \DI\ContainerBuilder();
$container = $builder->build();


$a = $container->get('A');
$a_b = $a->get_B();
$b = $container->get('B');
var_dump($a_b === $b); # true

class A{
    private $_b;
    function __construct(B $b)
    {
        $this->_b = $b;
        // $b->hello();
    }

    function get_B()
    {
        return $this->_b;
    }
}

class B{
    function __construct()
    {
        echo "Hello World! B";
    }
    function hello()
    {
        echo "B:say hello";
    }
}
