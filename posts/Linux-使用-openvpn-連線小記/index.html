# 前面可以用 VM 開啟，我這邊是用 vagrant
# 安裝 openvpn
sudo apt install openvpn tmux

# 下載 openv，可從 vpn gate 那邊下載
wget https://..../...ovpn 這邊下載檔名不能跟下載不一樣，建議還是用瀏覽器下載

# 開啟 tmux ，因為連接需要多開，
tmux 

# 執行 openvpn
openvpn ./xxxx.ovpn

# 切回 command line
