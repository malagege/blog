# 沒有環境變數，為true
if [ -z "$MODE" ]; then
    php .....
else
    # 時間用 0 0 28-31 * *
    # 日期加一天為1，就算是今日事月底最後一天
    DAY=$(date -d "$(date +%Y-%m-%d) +1 day" +%e)
    if [ "$MODE" = 'monthoflastday' ]; then
        if test $DAY -eq 1; then
            #  run the script here
            php ......
        fi;
    fi;
fi;

簡單說明就是 Rancher 要跑的 JOB 要多一個 MODE 環境變數
透過這個知道哪個要執行月底執行，有些不用

crontab 簡單判斷方法

55 23 28-31 * * [[ “$(date –date=tomorrow +%d)” == “01” ]] && myjob.sh
